CREATE TABLE PERSON (
  ID int(10) unsigned NOT NULL AUTO_INCREMENT,
  FIO varchar(150) NOT NULL DEFAULT '',
  PHONE varchar(12) NOT NULL DEFAULT '',
  EMAIL varchar(100) NOT NULL DEFAULT '',
  BIRTHDATE varchar(100) NOT NUll DEFAULT '',
  GENDER varchar(20) Not NULL DEFAULT '',
  BIOGRAFY varchar(300) NOT NULL DEFAULT '',
  CH boolean NOT NULL DEFAULT FALSE,  (CH - это проверка согласия
  PRIMARY KEY(ID)
);

CREATE TABLE LANG(
  LANG_ID int(10) unsigned NOT NULL AUTO_INCREMENT,
  LANG_NAME varchar(30) NOT NULL DEFAULT '',
  PRIMARY KEY(LANG_ID)
);


CREATE TABLE PERSON_LANG (
ID int(10) unsigned NOT NULL AUTO_INCREMENT,
U_ID int(10) unsigned Not NUll DEFAULT 0,
LANG_ID int(10) unsigned Not NUll DEFAULT 0,
PRIMARY KEY(ID)
);

CREATE TABLE PERSON_LOGIN(
  login VARCHAR(255) NOT NULL references LOGIN(login),
  ID int(10) unsigned NOT NULL AUTO_INCREMENT,
  FIO varchar(150) NOT NULL DEFAULT '',
  PHONE varchar(12) NOT NULL DEFAULT '',
  EMAIL varchar(100) NOT NULL DEFAULT '',
  BIRTHDATE varchar(100) NOT NUll DEFAULT '',
  GENDER varchar(20) Not NULL DEFAULT '',
  BIOGRAFY varchar(300) NOT NULL DEFAULT '',
  CH boolean NOT NULL DEFAULT FALSE,
  PRIMARY KEY(ID)
);

CREATE TABLE LOGIN (
  login VARCHAR(255) NOT NULL UNIQUE PRIMARY KEY,
  password VARCHAR(255) NOT NULL
);
